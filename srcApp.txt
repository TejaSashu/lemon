import React, { useState, useEffect } from 'react';

function App() {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  // Function to fetch data from an API
  const fetchData = async () => {
    try {
      const response = await fetch('https://jsonplaceholder.typicode.com/todos/1'); // Example API
      if (!response.ok) {
        throw new Error('Network response was not ok');
      }
      const result = await response.json();
      setData(result);
      setLoading(false);
    } catch (error) {
      setError(error.message);
      setLoading(false);
    }
  };

  // useEffect to call fetchData on component mount
  useEffect(() => {
    fetchData();
  }, []);

  return (
    <div className="App">
      <h1>API Data Fetch Example</h1>
      {loading && <p>Loading...</p>}
      {error && <p>Error: {error}</p>}
      {data && <div>
        <p><strong>ID:</strong> {data.id}</p>
        <p><strong>Title:</strong> {data.title}</p>
        <p><strong>Completed:</strong> {data.completed ? 'Yes' : 'No'}</p>
      </div>}
    </div>
  );
}

export default App;
